\newcommand\jumpbox[1]{
  \tikz[baseline=(n.base)]{
	  \node(n)[inner sep=1pt]{$#1$};
	  \draw[line cap=round]
	    (n.south west)
	    --(n.north west)
	    --(n.north east)
	    --(n.south east);}}

\newcommand\tacklebox[1]{
  \tikz[baseline=(n.base)]{
    \node(n)[inner sep=1pt]{$#1$};
    \draw[line cap=round]
      ($ (0, 0.25em) + (n.north) $)
      -- ($ (0.2em, -0.1em) + (n.south east) $)
      .. controls ($ (-0.35em,0.1em) + (n.south) $)
      .. cycle;}}

\newcommand\steppy[1]{
  \tikz[baseline=(n.base)]{
    \node(n)[inner sep=1pt]{$#1$};
    \draw[line cap=round]
      (n.south west) parabola 
      bend ($ (0.4em,0.3em)+(n.north west) $)
      ($ (0.1em,0)+(n.north) $);}}
